Total biomass

model_traits_total_biomass <- lmer(total_biomass ~  treatment * amax +
                                                        treatment * wue  +
                                                        treatment * sla +
                                                        treatment * pnue +

                                                        nfixer * amax +
                                                        nfixer * wue  +
                                                        nfixer * sla +
                                                        nfixer * pnue +

                            (1 |spcode) +

                            # Control over different plant heights
                            init_height,

                        data = data_for_regression_scaled_traits)
